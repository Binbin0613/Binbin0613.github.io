二、torch.nn.Module 对所有的神经网络构成最基本的类

import torch.nn as nn
import torch.nn.functional as F

class Model(nn.Module):
    def __init__(self):
        super().__init__()
        self.conv1 = nn.Conv2d(1, 20, 5)
        self.conv2 = nn.Conv2d(20, 20, 5)

    def forward(self, x):
        x = F.relu(self.conv1(x))
        return F.relu(self.conv2(x))

上述代码当中的神经网络包括两个过程：forward函数对x首先进行卷积1，卷积1的处理过程在init函数当中定义nn.Conv2d(1, 20, 5)，然后对其进行非线性的处理relu，再来一次卷积2和非线性relu，最后返回结果。

三、卷积操作

# 可以手动算一算
import torch
import torch.nn.functional as F

input = torch.tensor([[1, 2, 0, 3, 1],
                      [0, 1, 2, 3, 1],
                      [1, 2, 1, 0, 0],
                      [5, 2, 3, 1, 1],
                      [2, 1, 0, 1, 1]])

kernel = torch.tensor([[1, 2, 1],
                       [0, 1, 0],
                       [2, 1, 0]])
print(input.shape)
print(kernel.shape)
# torch.Size([5, 5])
# torch.Size([3, 3])

# 进行变换为四维的矩阵
input = torch.reshape(input, (1, 1, 5, 5))
kernel = torch.reshape(kernel, (1, 1, 3, 3))

print(input.shape)
print(kernel.shape)
# torch.Size([1, 1, 5, 5])
# torch.Size([1, 1, 3, 3])

# conv2d：表示2维卷积     conv3d：表示3维卷积
# conv2d表示进行卷积操作 stride为步进值
output = F.conv2d(input, kernel, stride=1)
print(output)
# tensor([[[[10, 12, 12],
#           [18, 16, 16],
#           [13,  9,  3]]]])

output2 = F.conv2d(input, kernel, stride=2)
print(output2)
# tensor([[[[10, 12],
#           [13,  3]]]])

# padding 表示在输入矩阵的外围填充一圈,默认填充的是0
output3 = F.conv2d(input, kernel, stride=1, padding=1)
print(output3)
# tensor([[[[ 1,  3,  4, 10,  8],
#           [ 5, 10, 12, 12,  6],
#           [ 7, 18, 16, 16,  8],
#           [11, 13,  9,  3,  4],
#           [14, 13,  9,  7,  4]]]])

四、卷积数据集

import torch
import torchvision
from torch import nn
from torch.nn import Conv2d
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

dataset = torchvision.datasets.CIFAR10("./data", train=False, transform=torchvision.transforms.ToTensor(), download=True)

dataloader = DataLoader(dataset, batch_size=64)

class BinConv2d(nn.Module):
    def __init__(self):
        super(BinConv2d, self).__init__()
        self.conv1 = Conv2d(in_channels=3, out_channels=6, kernel_size=3, stride=1, padding=0)

    def forward(self, x):
        x = self.conv1(x)
        return x

binConv2d = BinConv2d()
print(binConv2d)

step = 0
writer = SummaryWriter("logs")

for data in dataloader:
    imgs, targets = data
    output = binConv2d(imgs)    # 将图像放入到定义好的卷积处理流程当中进行处理
    print(imgs.shape)
    print(targets.shape)
    # torch.Size([64, 3, 32, 32])
    writer.add_images("input", imgs, step)
    # torch.Size([64, 6, 30, 30]) -> [xxx, 3, 30, 30]
    # 原始的output格式torch.Size([64, 6, 30, 30])，但是需要转化为[xxx, 3, 30, 30]
    output = torch.reshape(output, (-1, 3, 30, 30))
    writer.add_images("output", output, step)
    step = step + 1

writer.close()

五、池化层Pooling layers

maxpooling: 最大池化的作用：保留原始的特征的同时，也需要降低维度。换而言之就是对其进行压缩，我们看的视频是1080P的，池化之后是720p也可以满足对视频的需求，但是其降低了数据的计算量。

import torch
import torch.nn.functional as F
import torchvision
from torch import nn
from torch.nn import MaxPool2d
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

dataset = torchvision.datasets.CIFAR10("./data", train=False, download=True,
                                       transform=torchvision.transforms.ToTensor())

dataloader = DataLoader(dataset, batch_size=64)

# input = torch.tensor([[1, 2, 0, 3, 1],
#                       [0, 1, 2, 3, 1],
#                       [1, 2, 1, 0, 0],
#                       [5, 2, 3, 1, 1],
#                       [2, 1, 0, 1, 1]], dtype=torch.float32)
#
# input = torch.reshape(input, (-1, 1, 5, 5))
# print(input.shape)

class BinMaxPool(nn.Module):
    def __init__(self):
        super(BinMaxPool, self).__init__()
        # ceil_mode 表示：如果匹配的过程中有空洞的地方，是否需要跳过
        # kernel_size 表示： 每一次池化核走的时候，走3步
        self.maxpool1 = MaxPool2d(kernel_size=3, ceil_mode=False)

    def forward(self, input):
        output = self.maxpool1(input)
        return output


binmaxpool = BinMaxPool()
# output = binmaxpool(input)
# print(output)

writer = SummaryWriter("./logs_maxpool")
step = 0

for data in dataloader:
    imgs, targets = data
    # setp对应上每一步的输入和输出图像
    writer.add_images("input", imgs, step)
    output = binmaxpool(imgs)
    writer.add_images("output", output, step)
    step = step + 1

writer.close()

六、非线性激活relu处理和sigmoid处理

非线性激活函数实际上是给模型增强非线性的表达能力或者因素，有了非线性函数模型的表达能力就会更强。整个模型就像活了一样，而不是像机器一样只会做一些单一的线性操作。
没有激活函数的神经网络实际上是线性可加的，那么多线性层其实可以归为一层，只具有线性的神经网络表达能力及其有限。

注意：批的大小就是batch_size，每取一次就是一个step，step与batch_size是有关系的，因为通常数据集都很大，没有办法一次性全部取出来，单个单个的取浪费时间（与IO操作有关系），所有按照批处理的方式进行。

import torch
import torchvision
from torch import nn
from torch.nn import ReLU, Sigmoid
from torch.utils.data import DataLoader
from torch.utils.tensorboard import SummaryWriter

input = torch.tensor([[1, -0.5],
                      [-1, 3]])

input = torch.reshape(input, (-1, 1, 2, 2))
print(input.shape)   # 将数据转化为满足格式的

dataset = torchvision.datasets.CIFAR10("./data", train=False, download=True,
                                       transform=torchvision.transforms.ToTensor())

dataloader = DataLoader(dataset, batch_size=64)

class BinReLU(nn.Module):
    def __init__(self):
        super(BinReLU, self).__init__()
        self.relu1 = ReLU()
        self.sigmoid11 = Sigmoid()

    def forward(self, input):
        output = self.sigmoid11(input)
        return output


binRelu = BinReLU()
output = binRelu(input)
print(output)


writer = SummaryWriter("./logs_relu")
step = 0
for data in dataloader:
    imgs, targets = data
    writer.add_images("input", input, global_step=step)
    output = binRelu(imgs)
    writer.add_images("output", output, step)
    step += 1

writer.close()

八、损失函数与反向传播

损失函数：指的是实际的输出与目标之间的差距
方向传播：指为我们更新数据提供了一定的依据

import torch
from torch.nn import L1Loss
from torch import nn

# 注意的是：这里的[1, 2, 3]是指3个维度，是一维1， 2， 3
input = torch.tensor([1, 2, 3], dtype=torch.float32)
target = torch.tensor([1, 2, 5], dtype=torch.float32)

input = torch.reshape(input, (1, 1, 1, 3))
target = torch.reshape(target, (1, 1, 1, 3))

loss = L1Loss()
result = loss(input, target)

# 计算均方误差
loss_mse = nn.MSELoss()
result_mse = loss_mse(input, target)

print(result)       # tensor(0.6667)
print(result_mse)   # tensor(1.3333)


# 下面进行交叉熵损失函数的计算
x = torch.tensor([0.1, 0.2, 0.3])
y = torch.tensor([1])
x = torch.reshape(x, (1, 3))
loss_cross = nn.CrossEntropyLoss()
result_cross = loss_cross(x, y)
print(result_cross)     # tensor(1.1019)

交叉熵损失函数的计算

交叉熵损失函数的解读：

    假设：目标为检测左侧的图像，中间的为深度神经网络的模型图，分成三组[person, dog, cat]，其下标分别表示[0, 1, 2]输出的结果为[0.1, 0.2, 0.3], 目标是检测为dog[1]。

对应于公式中：x表示output，class表示target的值

则Loss(x, class) = -0.2 * 1 + log(exp(0.1) + exp(0.2) + exp(0.3))
