¶一、第一种方式使用GPU训练模型

¶方法：在损失函数，网络模型，数据(imgs, targets) 上添加 .cuda()
import torch
import torchvision
import time
from torch.utils.tensorboard import SummaryWriter

# 1、准备数据集(参数当中需要将其转换为tensor类型)
from torch import nn
from torch.utils.data import DataLoader

train_data = torchvision.datasets.CIFAR10(root="./data", train=True, transform=torchvision.transforms.ToTensor(),
                                          download=True)

test_data = torchvision.datasets.CIFAR10(root="./data", train=False, transform=torchvision.transforms.ToTensor(),
                                         download=True)

# 2、查看训练数据集 测试数据集的大小
train_data_size = len(train_data)
test_data_size = len(test_data)
# 下面是字符串的替换方法
print("训练数据集的大小为：{}".format(train_data_size))
print("测试数据集的大小为：{}".format(test_data_size))
# 训练数据集的大小为：50000
# 测试数据集的大小为：10000


# 3、使用DataLoader加载数据集
train_dataloader = DataLoader(train_data, batch_size=64)
test_dataloader = DataLoader(test_data, batch_size=64)


# 4、引入搭建好的神经网络： model.py
# 使用了 from model import *
class Bin(nn.Module):
    def __init__(self):
        super(Bin, self).__init__()
        self.model = nn.Sequential(
            nn.Conv2d(3, 32, 5, 1, 2),
            nn.MaxPool2d(2),
            nn.Conv2d(32, 32, 5, 1, 2),
            nn.MaxPool2d(2),
            nn.Conv2d(32, 64, 5, 1, 2),
            nn.MaxPool2d(2),
            nn.Flatten(),
            nn.Linear(64*4*4, 64),
            nn.Linear(64, 10)
        )

    def forward(self, x):
        x = self.model(x)
        return x
bin = Bin()
# 将网络模型转移到cuda上去
if torch.cuda.is_available():
    bin = bin.cuda()

# 5、构建损失函数
loss_fn = nn.CrossEntropyLoss()
if torch.cuda.is_available():
    loss_fn = loss_fn.cuda()


# 6、构建优化器
# 学习速率
# learning_rate = 0.01
# 也有下面的写法
# 1e-2 = (10) ^ (-2) = 0.01
learning_rate = 1e-2
optimizer = torch.optim.SGD(bin.parameters(), lr=learning_rate)

# 7、设置训练网络当中的一些参数
# 训练的次数
total_train_step = 0
# 测试的次数
total_test_step = 0
# 训练的轮数
epoch = 10

# 8、循环进行训练
# 在训练的过程当中添加tensorboard
writer = SummaryWriter("./logs_train")

for i in range(epoch):
    print("第 {} 轮训练开始".format(i+1))

    # 训练步骤开始
    bin.train()
    for data in train_dataloader:
        imgs, targets = data
        if torch.cuda.is_available():
            imgs = imgs.cuda()
            targets = targets.cuda()
        output = bin(imgs)
        loss = loss_fn(output, targets)

        # 优化器优化模型
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        total_train_step = total_train_step + 1
        if total_train_step % 100 == 0:
            print("训练次数: {} ， loss: {}".format(total_train_step, loss))
            # 将训练过程当中的额loss添加到tensorboard上
            writer.add_scalar("train_loss", loss.item(), total_train_step)


    # 测试步骤开始
    bin.eval()
    total_test_loss = 0
    total_accuracy = 0
    # 不需要进行梯度处理
    with torch.no_grad():
        for data in test_dataloader:
            imgs, targets = data
            if torch.cuda.is_available():
                imgs = imgs.cuda()
                targets = targets.cuda()
            outputs = bin(imgs)
            loss = loss_fn(outputs, targets)
            total_test_loss = total_test_loss + loss.item()
            accuracy = (outputs.argmax(1) == targets).sum()
            total_accuracy = total_accuracy + accuracy
    print("整体测试集上的loss: {}".format(total_test_loss))
    print("整体测试集上的正确率： {}".format(total_accuracy/test_data_size))
    writer.add_scalar("test_loss", total_test_loss, total_test_step)
    writer.add_scalar("test_accuracy", total_accuracy/test_data_size, total_test_step)
    total_test_step = total_test_step + 1

    # 在这里保存每一轮训练的结果
    torch.save(bin, "bin_{}.pth".format(i))
    print("模型已保存")

writer.close()

import torch
import torchvision
import time
from torch.utils.tensorboard import SummaryWriter


# 1、准备数据集(参数当中需要将其转换为tensor类型)
from torch import nn
from torch.utils.data import DataLoader

# 定义训练的设备
# device = torch.device("cpu")
# device = torch.device("cuda:0")
# device = torch.device("cude" if torch.cuda.is_available() else "cpu") 语法糖的格式
device = torch.device("cuda")

train_data = torchvision.datasets.CIFAR10(root="./data", train=True, transform=torchvision.transforms.ToTensor(),
                                          download=True)

test_data = torchvision.datasets.CIFAR10(root="./data", train=False, transform=torchvision.transforms.ToTensor(),
