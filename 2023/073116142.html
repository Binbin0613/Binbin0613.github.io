¶C++基础03：程序的基本结构和语句

# include<iostream>
# include<stdio.h>
#include"temp2.h"

using namespace std;

// C语言程序的基本结构和语句
int main() {

	// 一、C语言的语句和程序的基本结构
	// 1, 语句的分类
	//		（1）控制语句：共有9中控制语句if for while do break switch continue goto return
	// 	    （2）函数调用语句： 例如printf("Hello Word!");
	//		（3）表达式语句：  例如 int a; a = 5; a += 5;
	// 	    （4）复合语句：使用{}括起来的语句
	//					   凡是满足该条件的，{}内的语句都会执行
	//					   {
	//							int a = 1;
	//							......
	//						}   大括号后面不用加分号
	// 2、程序的三种基本结构
	//		（1）顺序结构
	//		（2）选择结构if-else/多分支选择结构switch
	//		（3）循环结构
	//					当型循环结构：先判断是否可行，再进行操作
	//					直到型循环结构：先执行操作，再判断条件是否可行，要强调的是，至少会执行一次操作
	// 
	// 3、赋值语句的特殊写法
	//		赋值语句可以编写在其他语句当中。


	// 多分支语句switch
	int i = 3;
	//switch当中为表达式，一般是整型变量或表达式
	//每一个case之间的顺序没有影响
	// 每一个case后的值不相同，不要忘记break（包括default后面），会导致继续执行后续语句，直到遇到break或整个switch执行完毕
	// 每一个case下面多行语句均会执行，如果没有加{}，每一条语句任会执行。
	// switch可以没有default。
	switch (i) {
		case 1: {
			printf("i=%d", 1);
			// ...
			break;
		}
		case 2:
		case 3:
		case 4: {
			printf("i=2或3或4");
			// ...
			break;
		}
		case 5: {
			printf("i=%d", 5);
			// ...
			break;
		}
		default: {
			printf("i=%d", -1);
			break;
		}
	}


	// 一、数据的输出
	// #include 表示预编译命令，作用是将一些文件包含到用户的源文件当中来，也就是将某些文件的内容原封不动的粘贴到#include这个位置
	// #include<stdio.h> 标准的I/O库，.h被称为头文件（head）
	// #include"stdio.h" 也是可行的。
	// 上述的两种包含头文件的区别： <>:表示从系统的目录当中进行查找文件，像加入标准的库文件使用<>，如果系统中没有，当前目录有，则会显示错误
	//								"":表示首先从当前的目录文件进行查找文件，如果找不到，再到系统当中查找。""用于对自己编写的头文件的引入，让系统优先使用当前目录当中的定义的头文件

	char a, b, c;
	a = 'F';
	b = 'T';
	c = 'A';
	putchar(a);
	putchar(b);
	putchar(c);

	// printf输出格式
	// %d: 以10进制输出一个数字
	// %o: 以8进制输出一个数字
	// %x: 以16进制输出一个数字
	// %u: 以10进制输出一个无符号数字
	// %c: 输出一个字符
	// %s: 输出一个字符串
	// %f: 输出一个实数
	// %.nf: 输出一个实数，保留n位小数

	// 数据的输入与输出
	char temp;
	temp = getchar();		// 从键盘上输入一个字符
	putchar(temp);			// 输出字符temp

	// scanf: 格式化输入函数，用来输入任何类型的多个数据
	// scanf(格式控制，地址表列)
	// 当输入回车时，表示输入结束
	int x, y, z;
	scanf_s("%d%d%d", &x, &y, &z);//  表示在xyz的地址上输入数据,多个输入之间可以使用空格，回车，tab间隔
	printf("%d%d%d", x, y, z);

	// 关系运算符（<>!=<=>=）与关系表达式
	// 关系运算符的优先级  小于  算数运算符的优先级
	// 关系运算符的值，只有真和假两种

	// 逻辑运算符与逻辑表达式(||或   &&与   !非)
	// 用逻辑运算符 将 关系表达式 连接起来的，就是逻辑表达式
	// 逻辑表达式的值也是真或假
	// || && ： 双目运算符
	// ! : 单目运算符
	// 注意的是： 千万不要把&& 和|| 写成 & |,不会报错，但是逻辑会出错


	// 逻辑运算符求值问题:在逻辑表达式求解中，不是所有逻辑运算符都会被执行，
	// 只有在必须执行下一个逻辑运算符才能求出表达式的结果时，才执行该运算符。
	// a && b && c 只有a为真(非0)）才需要判断b，只有a, b都为真才判断c·只要a为假，就不必判断b。如果a为真，b为假，就不必判断c
	int m = 0;
	if ((3 > 5) && (m = 8)) {	// 此时，后面m=8这一表达式不会被执行
		printf("%d",m);
	}

	// 循环结构语句（当型循环结构，直到型循环结构）
	// goto, while, do while, for
	// goto语句: 无条件转向语句，用来跳转到某个程序位置进行执行
	// goto 语句标号； （其中语句标号是一个标识符，只能有字母数字下划线组成，且第一个字符必须是字母或下划线）
	// goto语句的用途： （1） 与if语句构成循环结构
	//					（2）从循环结构当中跳出，不推荐，破坏循环结构
	// 案例： 使用goto语句实现1~100的加法
	int number = 1; int sum = 0;
	loop:
		if (number <= 100) {
			sum = sum + number;
			number++;
			goto loop;
		}
	printf("%d", sum);
	// 不建议使用goto语句


	// while 循环结构(先判断循环条件，再执行内容)

	// do while循环结构（直到型循环结构）
	// 直到型循环结构，至少会执行循环体内的语句一次


	// for:循环语句
	// for语司的一般形式 :
	// for(表达式1 : 表达式2; 表达式3)内嵌的语句
	// 执行过程:
	// a)先求解表达式1的值
	// b)再求解表达式2的值的
		// b.1)若表达式2的值为真，则执行for语句中指定的内嵌语句﹐再求解表达式3的值，反复循环步骤2，一直到表达式2的值为假。
		// b.2)若表达式2的值为假，则整个for循环结束，程序流程跳转到for语句后边的语句去执行。
		
		// 这里我们要特别注意:表达式1的值只会被执行一次。
		// for语句最简单也是最常用的引用形式 :
		// for (循环变里赋初值 : 循环变里结束条件; 循环变里增加值）内嵌的语句

	// 特殊情况
	// 1、第一个表达式可以省略，但是循环变量需要在事前所定义
	// 2、第二个表达式可以省略，需要在循环体内编写break语句用于中断循环。
	// 3、第三个表达式可以省略，i++可以放在循环体内

	// goto while do while for 循环之间的比较
	// 1、大多数情况下，这些循环之间是可以互相替换的
	// 2、不建议使用goto语句,会破坏结构化程序设计
	// 3、while和for循环是先判断后执行语句，dowhile是先执行语句，后进行判断
	// 4、三种循环语句都可以使用break中断循环
	// 5、continue表示退出当前循环，会进行后续的循环内容
	// 6、 continue只能够用于while do while for三种结构体当中，不能用于switch
    // 7、contine表示跳过下面未执行的语句，接着执行下一次循环
    for(int i = 1; i<=100;i++){
        if(i % 3 == 0){
            continue;
        }
        printf("%d ", i);
    }
	return 0;
}
