C++基础07：指针

#include<iostream>
using namespace std;


// 一、预备知识
	// 静态存储区，动态存储区。。。不同的变量会存储在不同的存储区当中
	// 1、有些变量，会在编译的时候分配内存
	// 2、有些变量，会在程序运行的时候分配内存

// 二、直接访问和间接访问
	// 直接访问： 按照这个变量的地址取值
	// 间接访问： 将变量i的地址存放到另一个存储单元中
		// 我们通常用int char等类型来定义并存储变量的值
		// 同时也有一种特殊的变量，能够存储另一个变量的“地址”，这就是指针变量 point = &i；
		// 有时候，地址也被称为指针。

// 三、指针变量的定义
	// 类型标识符 *标识符 int *point; 这个指针变量 指向整型标量
	// 一个指针变量只能指向同一类型的变量
	
// 四、指针变量的引用
	// 指针变量中只存放地址，不要将整型变量赋值给一个指针变量
	// *p = i

// 五、坑爹的解析
	// &*p ： & * 两个运算符的优先级相同，&（*p） = &i = p
	// *&i : *&p = *(&i) = *p = i
	// (*p)++ : i++
	// *p++ ：*和++的优先级相同，*p++ = *(p++) 这里表示p所指向的内存 + p类型所占内存数后的地址
		// &i = p = 1000;  p++ = 1000 + sizeof(int)4; 但是这一块内存不被p所使用,谨慎使用p++


// 六、分析一下,为什么下面会出问题
void swap(int *p1, int *p2) {
	// 这个过程是： 
	int temp;
	temp = *p1;		// *p1 = a = 1 ==> temp =1
	*p1 = *p2;		// *p2 = b = 2 ==> *p1 = 2    a = 2
	*p2 = temp;		// *p2 = 1

	// 下面这种方式会出现问题 有点懵逼，但是就是下面这种方式不行
	// int *temp;
	// temp = p1;
	// p1 = p2;
	// p2 = temp;
}

// 七、数组的指针
	// 数组指针： 数组的首地址  
	// 数组元素的指针： 数组元素的地址

// 八、改变数组元素的几种方式（形参与实参的对应）
// (1) 实参和形参都是数组名称
void changeValue1(int tempArr[]) {
	tempArr[0] = 100;
}
// (2) 实参用数组名，形参用指针变量
void changeValue2(int* p) {
	p[1] = 110;	// p[1] = *(p+1)
}
// (3) 实参用指针变量，形参用指针变量
void changeValue3(int* p) {
	p[2] = 120;	// p[1] = *(p+1)
}
// (4) 实参用指针变量，形参用数组名
void changeValue4(int tempArr[]) {
	tempArr[3] = 130;
}

// 九、指针数组和数组指针


// 十、字符串指针和指向字符串的指针变量
	//char mystr1[] = "I love China!";
	//char mystr2[] = "I love China!"; 
	//printf("%s\n", mystr1);			// 将字符串拷贝到字符数组中
	//printf("%s\n", mystr2);			// 这两个变量的地址不同
	//
	//char *mystr1 = "I love China!";	// 这两个变量的相同，都指向 "I love China!";的地址
	//char *mystr2 = "I love China!";	// 不是拷贝
// C语言对字符串常量有特殊的处理，在内存中开辟一段类似于字符数组的空间来存储字符串常量
// 所有 "I love China!";在内存中，有一块自己的内存地址。


int main() {

	// 指针的定义
	int i;
	int *pointToi;	// 定义一个指针变量
	pointToi = &i;	// 表明了pointToi指向了整型变量i


	// 将p1指向大的数
	/*int a = 1;
	int b = 2;
	int *p1;
	int *p2;
	if (a < b) {
		swap(*p1, *p2);
	}*/

	//只要是数组，数组之间的地址都是挨着的
	int arr[5];
	arr[0] = 1; arr[1] = 2; arr[2] = 3; arr[3] = 4; arr[4] = 5;
	int* p = &arr[0];
	*p = 10;			// 表示改变了*p = arr[0] 的值
	p = p + 1;			// 由于数组之间的地址是连续性的，所以p+1表示p所指向的元素的下一个元素
	// p指向数组首元素地址，p + i 表示p指向了数组a[i]. a + i也是表示元素a[i]

	// ！！！！！注意！！！！！
	// a+i	  表示数组第i个元素a[i]  a就是数组的初始地址 
	// p+i	  表示数组第i个元素a[i]	 p指向数组的初始地址
	// *(p+i) 表示数组第i个元素a[i]	 p可以++ p++合法，就是往后加一
	// *(a+i) 表示数组第i个元素a[i]	 a不可以++ a++不合法
	// p[i]   表示数组第i个元素a[i]


	// 八、改变数组元素的4种方式（形参与实参之间的4种对比方式）
	int a[5] = { 1,2,3,4,5 };
	// (1) 实参和形参都是数组名称
	changeValue1(a);		// changeValue1的形参和实参公用某一块内存
	printf("%d\n", a[0]);	// 100 当数组名称作为实参传递时，传递的是数组的地址，实参数组原本的内容也会在changeValue1中改变
	// (2) 实参用数组名，形参用指针变量
	changeValue2(a);		// 实参是指针变量，并且又明确的指向
	printf("%d\n", a[1]);	// 110 
	// (3) 实参用指针变量，形参用指针变量
	int *pa = a;
	changeValue3(pa);		
	printf("%d\n", a[2]);
	// (4) 实参用指针变量，形参用数组名
	int *paa = a;
	changeValue4(paa);
	printf("%d\n", a[3]);
	// 总结： 数组名称（即数组的地址）与指针可以相互替换


	// 注意的是： 二维数组的aaa[1] 就是aaa[1][0]的地址 第二行元素的首地址
	int aaa[2][2] = { {1,2},{3,4} };
	printf("%d\n", aaa[1]);		// 下面两个元素的值相同
	printf("%d\n", &aaa[1][0]);


	// 实践一下
	int aa[2][3] = { 86,86,86,86,86,86 };
	int *pp = (int *)(aa + 1);	// 将二维数组的第二行首地址给p  aa[1][0]。。。aa[1][2]
	*pp = 100;					// aa[1][0] = 100
	pp++;						// 走4个字节 p指向 a[1][1]
	*pp = 110;					// aa[1][1] = 110;
	
	// 九、指针数组和数组指针


	return 0;
}

​ 2、有些变量，会在程序运行的时候分配内存

二、直接访问和间接访问

直接访问： 按照这个变量的地址取值
间接访问： 将变量i的地址存放到另一个存储单元中
我们通常用int char等类型来定义并存储变量的值
同时也有一种特殊的变量，能够存储另一个变量的“地址”，这就是指针变量 point = &i；
有时候，地址也被称为指针。

三、指针变量的定义

​ 类型标识符 *标识符 int *point; 这个指针变量 指向整型标量
​ 一个指针变量只能指向同一类型的变量
​

四、指针变量的引用

指针变量中只存放地址，不要将整型变量赋值给一个指针变量
*p = i

五、坑爹的解析

&p ： & * 两个运算符的优先级相同，&（*p） = &i = p
*&i ： *&p = *(&i) = *p = i
( *p)++ ： i++
// * p++ ： 和++的优先级相同，*p++ = *(p++) 这里表示p所指向的内存 + p类型所占内存数后的地址
&i = p = 1000; p++ = 1000 + sizeof(int)4; 但是这一块内存不被p所使用,谨慎使用p++
