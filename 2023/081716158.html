	// （2） 析构函数也是类的成员函数，没有返回值，没有参数，不会被重载，只能有唯一一个析构函数
	// （3） 在调出main函数时，对象会调用析构函数，上面创建了6个对象，所以上述会调用6次析构函数
	// （4） 一般而言对象内成员变量，先定义先初始化，先定义后销毁
	//

	Man* m1 = new Man();

	Human* h2 = new Man();
	//Man* m2 = new Human();  // 子类指针无法new父类对象
	//delete h1;
	delete m1;
	delete h2;


	// 八、虚函数
	// 具体需求： 我们声明一个父类的指针，使其能够访问父类的方法和子类的同名方法；
	// 处理方式： 在父类中，将这个同名函数使用virtual关键字修饰，声明为虚函数
	// 只要在父类中被声明为虚函数，那么在子类中不管加不加virtual，都是虚函数
	
	// new的是那个类（子类或者父类），后续调用的就是那个类的同名方法
	Human *h3 = new Man();	// 创建对象时，new的是哪种对象，访问的同名函数就是哪种对象的虚函数
	h3->eat();				// 男人吃饭，调用的是子类的虚函数eat
	delete h3;

	// override ： 用于在子类当中修饰虚函数的，表示对父类中函数的重写： void eat() override;
	// final：     用于对父类函数的修饰，被final修饰的父类函数无法被子类重写
	// 调用函数是 “动态绑定”的，动态： 只有程序执行的时候才知道调用的是那个子类的eat()函数


	// 九、多态性
	// 多态性是 只对虚函数来说的
	// 多态性： 体现在具有继承关系的父类和子类之间的，子类的重写的成员函数eat() 同时父类把eat声明为virtual
	// 程序运行时，通过父类的指针，动态绑定到父类指针上的对象（子类对象或父类对象）
		//  然后系统内部会插叙一个虚函数表，找到函数eat的入口地址，调用子类或父类的eat()，这就是运行时的多态性

	// 十、纯虚函数
	// 纯虚函数：在基类中声明的虚函数，在父类中没有定义，但要求所有派生类都要定义该虚函数自己的实现方法
	// 基类中的实现方法是在基类中的虚函数声明中（没有函数体，在后面加上 = 0）： virtual void eat() = 0

	// 注意：  一旦一个类中有了纯虚函数，那么你就不能生成这个类的对象了，这个类就是  抽象类
	// 抽象类 不能用来生成对象，主要目的是用来同一管理子类
	// 所以：  (1) 纯虚函数的类 叫 抽象类， 不能用来生成对象，主要用于当做基类生成子类的用处
			// (2) 子类中必须要实现基类中定义的纯虚函数

	//Human* h1 = new Human(); // 純虚函数不能生成父类的对象
	// 但是可以生成子类的对象，并且调用子类的纯虚函数
	Human* h4 = new Man();	
	h4->eat();

	// 十一、虚析构函数 : 基类的析构函数一般写成虚函数
	cout << "----------------" << endl;
	Human* pman = new Man();
	delete pman;					// 坏事了，没有调用子类的析构函数
	// 结论： 在用基类指针 new 子类的对象时，在delete的时候，系统不会调用派生类的析构函数，存在问题
	// 解决问题： 将父类的析构函数定义为虚函数

	// 记住： 在public继承关系中，基类对派生类及其对象的操作，智能影响到从基类继承下来的成员
	// 如果想要把基类对非继承成员进行操作，则也要把基类的这个函数定义为虚函数，析构函数也是如此
	// 另外就是，基类中析构函数的虚属性也会被继承给子类，这样的话子类中的析构函数也就成为了需函数，虽然名字不同

	//delcte phuman的时候，肯定是要调用父类的析构函数，但在父类析构函数中它要是想要调用子类Men的祈构函数，那么
	// Human这个类中的析构函数就要声明为virtual的，也就是说C++ 中为了获得运行时的多态行为，所调用的成员函的必须得是virtual的。

	//所以大家要记住结论:
	// 如果一个类，想要做“基类”，我们务必要把这个类的析构函数写成virtual析构函数;
	// 只要基类的析构函数是虚函数，就能够保证我们delete基类指针时能够运行正确的析构函数版本;
	// 也就是能把所有的对象delete干净，不会造成内存泄漏
	// 虚函数会增加内存的开销，类里面定义虚函数，编译器就会给这个类增加虚函数表，在这个表里存放虚函数的指针

	// 十二、友元函数
	/*Man m2;
	functemp(m2);*/
	// 


	return 0;

#ifndef __MYTIME__
#define __MYTIME__

// 注意：当前头文件是myTime类的声明（定义）
// 情况：在1.cpp和2.cpp文件当中均#include<myTime.h>，生成时，为什么系统不会报错
//       但是当在这个文件中声明（定义）一个全局变量
// int global_temp1 = 0;	  // 声明 并且定义了变量     如果多个源文件引用了本头文件，会报错
// int global_temp(int);   // 仅仅声明，没有定义变量  如果多个源文件引用了本头文件，不会报错
// 答案： 这是因为遵守： “单一定义规则One-Definition Rule ODR”  根据此规则， 如果对同一个类的两个定义完全相同且出现在不同编译单位，会被当作同一个定义
// 类不会出现这种情况
// https://www.apiref.com/cpp-zh/cpp/language/definition.html
// https://blog.csdn.net/baoxiaofeicsdn/article/details/48338515

class myTime {
public:
	int Hour;
	int Minute;
	int Second;

	// 成员函数的定义，成员函数的内容在temp10_类_time.cpp文件中
	void initTime(int tempHour, int tempMinute, int tempSecond);

	// 用于访问私有成员变量的函数
	void initMillTime(int mls);

	// 构造函数  
	// 一般这种带一个参数的成员函数，声明为explicit，不允许隐式的类型转换
	explicit myTime();		// 无参构造函数
	myTime(int tempHour, int tempMinute, int tempSecond);		// 三个参数的构造函数

	// 拷贝构造函数
	// 一般而言，在拷贝构造函数的第一个参数前面加上const
	myTime(const myTime &tempTime, int a = 10);

	// 重载运算符=
	myTime&  operator=(const myTime& tempObj);

	// 析构函数（释放函数）
	~myTime();

private:
	int Millsecond;

public:

	// 在类的声明头文件当中，添加函数的实现
	// 这种在类的定义中实现的成员函数，会被当做inline内联函数处理
	void addHour(int number) {
		Hour += number;
	}

	// 成员函数中this的使用
	myTime& add_Hour(int number);

	//声明静态成员变量
	static int myStatic;		// 没有分配内存，不能在这里初始化
	
	// 声明静态成员函数
	static void myStaticFun(int number);
};

#endif

二、析构函数:

相对于构造函数： 对象在销毁的时候，会调用析构函数

（1） 如果我们不写析构函数，编译器会给定一个默认的析构函数，默认的析构函数为空，没干啥事
（2） 析构函数也是类的成员函数，没有返回值，没有参数，不会被重载，只能有唯一一个析构函数
（3） 在调出main函数时，对象会调用析构函数，上面创建了6个对象，所以上述会调用6次析构函数
（4） 一般而言对象内成员变量，先定义先初始化，先定义后销毁

// 使用new delete创建和销毁对象,以下的代码必须要成对出现
myTime *myTime7 = new myTime();
delete myTime7;		// 使用delete，系统会调用析构函数

对比与总结：
构造函数： 做了两件事情： 函数体之前（初始化列表） 和 构造函数体内
析构函数： 做了两件事情： 函数体之内 代码块（销毁其他内容） 和 函数体之后 销毁对象

查询： 父类中修饰的变量，被子类修饰继承，那么子类的访问权限是什么？

六、基类指针，派生类指针

// 传统的new对象的方式
//Human* h1 = new Human();
Man* m1 = new Man();
// 新玩法： 父类指针new一个子类对象
Human* h2 = new Man();
//Man* m2 = new Human();  // 子类指针无法new父类对象
//delete h1;
delete m1;
delete h2;

七、虚函数

具体需求： 我们声明一个父类的指针，使其能够访问父类的方法和子类的同名方法；

处理方式： 在父类中，将这个同名函数使用virtual关键字修饰，声明为虚函数
只要在父类中被声明为虚函数，那么在子类中不管加不加virtual，都是虚函数

// new的是那个类（子类或者父类），后续调用的就是那个类的同名方法
Human *h3 = new Man();	// 创建对象时，new的是哪种对象，访问的同名函数就是哪种对象的虚函数
h3->eat();				// 男人吃饭，调用的是子类的虚函数eat  ,如果在父类中不是定义的虚函数，则只能为"人类需要吃饭"
delete h3;

override ： 用于在子类当中修饰虚函数的，表示对父类中函数的重写： void eat() override;
final： 用于对父类函数的修饰，被final修饰的父类函数无法被子类重写
调用函数是 “动态绑定”的，动态： 只有程序执行的时候才知道调用的是那个子类的eat()函数

八、多态性

多态性是 只对虚函数来说的
多态性： 体现在具有继承关系的父类和子类之间的，子类的重写的成员函数eat() 同时父类把eat声明为virtual
程序运行时，通过父类的指针，动态绑定到父类指针上的对象（子类对象或父类对象，然后系统内部会插叙一个虚函数表，找到函数eat的入口地址，调用子类或父类的eat()，这就是运行时的多态性

九、纯虚函数

纯虚函数：在基类中声明的虚函数，在父类中没有定义，但要求所有派生类都要定义该虚函数自己的实现方法

基类中的实现方法是在基类中的虚函数声明中（没有函数体，在后面加上 = 0）： virtual void eat() = 0

注意： 一旦一个类中有了纯虚函数，那么你就不能生成这个类的对象了，这个类就是 抽象类

(1) 抽象类 不能用来生成对象，主要目的是用来同一管理子类

(2) 纯虚函数的类 叫 抽象类， 不能用来生成对象，主要用于当做基类生成子类的用处

(3) 子类中必须要实现基类中定义的纯虚函数

//Human* h1 = new Human(); // 純虚函数不能生成父类的对象
// 但是可以生成子类的对象，并且调用子类的纯虚函数
Human* h4 = new Man();	
h4->eat();

十、虚析构函数 : 基类的析构函数一般写成虚函数

Human* pman = new Man();
delete pman;					// 坏事了，没有调用子类的析构函数
// 结论： 在用基类指针 new 子类的对象时，在delete的时候，系统不会调用派生类的析构函数，存在问题
// 解决问题： 将父类的析构函数定义为虚函数----养成习惯

记住： 在public继承关系中，基类对派生类及其对象的操作，智能影响到从基类继承下来的成员
如果想要把基类对非继承成员进行操作，则也要把基类的这个函数定义为虚函数，析构函数也是如此
另外就是，基类中析构函数的虚属性也会被继承给子类，这样的话子类中的析构函数也就成为了需函数，虽然名字不同

delcte phuman的时候，肯定是要调用父类的析构函数，但在父类析构函数中它要是想要调用子类Men的祈构函数，那么
Human这个类中的析构函数就要声明为virtual的，也就是说C++ 中为了获得运行时的多态行为，所调用的成员函的必须得是virtual的

所以大家要记住结论:

如果一个类，想要做“基类”，我们务必要把这个类的析构函数写成virtual析构函数;
只要基类的析构函数是虚函数，就能够保证我们delete基类指针时能够运行正确的析构函数版本;
—也就是能把所有的对象delete干净，不会造成内存泄漏
虚函数会增加内存的开销，类里面定义虚函数，编译器就会给这个类增加虚函数表，在这个表里存放虚函数的指针

十一、友元函数

因为friend 友元函数不属于类成员函数，所以不受public private protected的限制
在类的申明头文件中添加： friend void func();

总结： func() 是一个函数，通过声明为某个类的友元函数，它就可以访问这个类的所有成员，包括public protected private

注意：

(1) 友元关系不能被继承，
(2) 友元关系是单向的
(3) 友元关系没有传递性
