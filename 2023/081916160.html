C++进阶10：类-左右值-左右值引用-move

#include<iostream>
#include<vector>
using namespace std;

int main() {

	// 一、左值和右值
	int i = 10; // 对象： 一块内存区域
	// 左值： “能用在赋值语句等号左侧的东西” ，它能够代表 一个地址
	// 右值： “不能作为左值的值就是右值”，右值不能出现在赋值语句的左侧

	// 结论:
	// C++的一条表达式，要么就是右值，要么就是左值，不可能两者都不是

	// 左值有的时候可以被当做右值使用
	i = i + 1;  // i是一个左值，不是右值，虽然出现在了等号右边
	// i用在了等号右边的时候，我们认为： i有一种右值属性（不是右值）
	// i用在了等号左边的时候，用的是i代表的内存中的地址，我们认为： i有一种左值属性

	// 用到左值的运算符有哪些？
	// (1) 赋值运算符
	(i = 50) = 20;		// 最终的结果i=10，因为赋值运算符的结果是一个左值

	// (2) 取地址符
	&i;

	// (3) string vector 下标[] 都需要左值
	string abc = "Bin";
	vector<int>::iterator iter;

	iter++; iter--;
	abc[1];

	// (4) 通过看一个运算符在一个字面值上能不能操作，我们就可以判断运算符是否用到的左值
	// i++

	// 注意：
	// 左值表达式 就是 左值， 右值表达式 就是右值
	// 左值： 代表一个地址，所以左值表达式的结果，就是一个对象，就得有地址
		// 求值结果为对象的表达式，不代表一定是左值，具体再分析。

	// 总结
	// 变量一定是左值
	// 100 这种常量 不是左值


	// 二、引用分类
	int value = 10;
	int& refval = value;
	refval = 20;
	// 三种形式的引用
	// (1) 左值引用 （绑定到左值）

	// (2) const引用（常量引用），也是左值引用。我们不希望修改值的对象
	const int& refval2 = value;
	//refval2 = 30;  // 非法，因为表达式的左值无法被修改

	// (3) 右值引用（绑定到右值）
	int&& refrightvalue = 3;		// 右值引用绑定到一个常量
	refrightvalue = 5;


	// 三、左值引用
	// 引用左值，绑定到左值上
	//char* p = nullptr;		// 指针有空指针的说法
	// 但是，引用没有空引用的说法，必须在定义的时候初始化-绑定左值

	int a = 1;
	int& b{ a };  // 把b绑定到a上去
	//int& c;     // 错误，引用必须要初始化
	// int &c = 1;		// 错误，左值引用不能绑定到右值，必须绑定到左值
	const int& c = 1;	// 可以，const引用可以绑定到右值，这一句等价于： int temp = 1; const int &c = temp;

	// 四、右值引用
	// 右值引用，就是引用右值，也就是说，必须绑定到右值。
	// &&： 希望用右值引用来绑定一些即将销毁或者是一些临时的对象上

	int&& refrightvalue2 = 10;  // 右值引用可以理解成为一个对象的名字


	// 总结：绑定到左值上的引用，一般不能绑定到右值，反之同理


	string str = "Bin";
	string& s = str;			// 可以，左值引用绑定左值
	// string& s2 = "Bin";		// 不可以，左值引用不能绑定到临时变量，临时变量被系统当做右值
	const string& s3 = "Bin";	// 可以，系统创建了一个临时变量，然后绑定到左值引用s3上去
								// const引用不但可以绑定到右值，还可以执行到string的隐式类型转换并将所得的值放到string临时变量中
	// string &&s4 = str;		// 不可以，右值不能绑定到左值
	string&& s4 = "Bin";		// 可以，绑定到一个临时变量，临时变量的值为: Bin

	int m = 10;
	int& m1 = m;				// 可以，左值引用绑定左值
	//int&& m2 = m;				// 不可以，不能将右值绑定到左值上
	int&& m3 = m * 100;			// 可以， i*100 是右值，右值引用能绑定到右值上
	const int&& m4 = m * 100;   // 可以，邦定到右值上


	// 总结：
	// 1、返回左值引用的函数，连同赋值，下标，解引用和前置递增减运算符(--i)，都是返回左值表达式的例子
	// 2、返回非引用类型的函数，连同算术，关系，位以及后置递增减运算符(i--),都生成右值，不能将左值引用绑定到这类表达式上


	// 解释一下i++ 和 ++i
	// ++i 左值表达式，  首先给变量i+1  然后返回i本身
	//  (++i) = 99   i被赋值为99， 先i = i+1 然后 i = 99

	// i++ 右值表达式
	// i++ 先产生一个临时变量temp， 记录i(_temp = i)用来使用，再给i+1 ,接着返回这个临时变量
	// (i++) = 99;    语法错误

	int i = 1;
	int&& r1 = i++;//成功绑定右值。但是此后r1的值和i没有关系。
	//i +=5;
	//int &r2 = i++;//不可以,不能绑到右值表达式上去
	int &r3 = ++i;  //r3绑到i，r3就变成i的别名了。
	//i += 5;
	//int &&r4= ++i; /不可以,右值不能绑定左值表达式。
	
	//重点强调:
		// (1)rl虽然是右值引用（绑定到了右值)，但是r1本身它是左值〈你要把rl看成一个变量)。因为它在=左边呆着
		int& r5 = r1;  //r1是左值。
		//int &&r6 = r1;
		// (2）所有变量，看成左值。因为他们是有地址的。而且你用右值引用也绑定不上。
		//int &&r11 = i;//失败。
		// (3）任何函数里边的形参都是左值。void f(int i, int &&w) :w是右值引用，但w本身是左值。
		// (4）临时对象都是右值。下节详细讲解临时对象。
			// (4.1) 右值引用的目的
				// (a) c++11引入， &&，代表一种新的数据类型
				// (b) 提高程序的运行效率，把拷贝对象变成移动对象，提升程序运行效率(在对象拷贝的过程中
				//      会将原对象的成员变量依次的拷贝到新的对象中去，但我们利用引用的方式将原对象的内存直接给新对象
				//		降低了内存的消耗。)
				// (c) 移动对象如何发生  && (应付移动构造函数，应付移动赋值运算符用的)
		//


	// 五、std::move函数
	// c++11标准库中的新函数
	// 由于在程序中可能与同名的move函数发生冲突，所以建议在使用的时候： std::move()
	// 实际上这个函数根本上就没有移动的操作
	// 作用： 将一个左值，强制转换为一个右值，带来的结果就是一个右值可以绑定上去
	// 注意： 在使用了std::move(n) 之后，后续就不要再使用n了
	int n = 10;
	//int&& ri = n;  // 错误，n是左值，无法绑定到右值引用上
	int&& ri = std::move(n);	// 可以,将n转换为右值，然后绑定到右值的引用上
	n = 20;
	// ri 和 i 表示同一块地址

	string bin = "BinBin";
	string de = std::move(bin);
	// 执行上面两条语句之后，bin内容变为空，de内容为”Binbin",但是注意这里不是move转移的操作
	// 而是在这个过程中执行了string中的移动构造函数


	// 六、左值右值总结说明
	return 0;
}
