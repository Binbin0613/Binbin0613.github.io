C++基础05：函数

#include<iostream>

using namespace std;

// 函数的内容
// 注意的是，如果函数写在了main函数的下面，则找不到该函数，需要事先声明
// 如果将函数写在main函数的上面，则表示函数以及定义并声明了

// 函数的定义
// 一个大项目通常不会存放到一个源文件里，因此一个C项目由一个或多个源文件组成
// 这些函数可以分别放到源文件里，被所有源文件公用
// 程序从main函数开始，也是从main函数当中结束
// 函数通常分为两类： 库函数  自定义函数

// 1、定义一个函数时，形式参数在函数被调用前不分配内存，调用的时候会分配内存，函数调用结束之后，形式参数的内存就被释放了，所以形式参数只能在函数内部使用
// 2、在函数调用的时候，实参的值自动赋值给了形参
// 3、在C语言当中，实参变量对形参变量的传递是“值传递”，也就是单向传递，只能由实参传递到形参
//	  即完成函数的调用之后，原本的实参的值并没有改变

// 函数的调用
// C语言所编写的函数，只能返回一个值
// 函数的调用有三种方式： 1 单独作为一条语句 2 函数出现在一个表达式中 3 函数作为另一个函数的实参

// 函数的声明
// 1、可以将函数写在main函数的前面，表示函数的定义并声明
// 2、我们必须将函数的声明放到任何源代码文件的具体函数之前（一般就是源代码的开头），才能保证这些函数调用其他函数时，被调用的函数OK
// 3、函数的声明：
	// 类型标识符 函数名(形参列表);

// 函数的嵌套定义
// C语言当中，不允许在一个函数当中定义另外一个函数
// C语言允许函数的嵌套调用

// 函数的递归调用： 在函数体内调用自己
// 函数的递归调用过程当中，只有一个函数结束之后，函数的局部变量占用的内存才会被释放
int getJieCheng(int number) {// 1*2*3*4*5
	if (number == 1) {	// 递归函数必须要有结束条件
		return 1;
	}
	return number * getJieCheng(number-1);// 递归关系
}
// 个人理解的递归函数的编写
// 在函数体内要确定所有的边界条件，通常是特殊情况我们能确定的
// 然后使用其中的递归条件直接重复调用函数（汉诺塔，斐波拉契数列，棋盘可移动位置）

// 递归的优缺点
// 代码精简，但是如果调用的层数太深，可能会导致调用栈（内存）溢出
// 效率和性能都不高，无法保证深层次的运行效率


// 数组元素/数组名作为函数的参数
// 数组元素作为函数的参数： int a[5]; a[1] = 1; getJieCheng(a[1]);
// 数组名作为函数的参数： 数组名是数组的首地址，传递的是数组的首地址,数组可以是多维数组
	// 此时，函数中的形参，也必须是数组名，或者是数组指针
// 注意：数组名作为函数参数时，不是“值传递（形参会分配内存）”，不是单向传递，而是把实参数组的开始地址传递给了形参数组（不会分配内存了）
	// 这样，两个数组就会共同占用同一块内存。这个叫地址传递，也就是说，形参数组的值发生变化，会导致
	// 实参数组当中的值发生变化
void testShuZuCanShu(int temp[]) {	// 数组名作为参数，调用这个函数，temp和tempA是同一块地址
	temp[0] = 100;
	temp[1] = 100;
	// 这个函数的形参temp的大小可以声明大小，但是没有作用，函数会将实参的数组的地址给你
	// temp[100],实际上temp只能调用tempA[2]两个内容，其他的内存块temp无法控制，不该temp管，不属于temp
	// temp[5] = 100; 这一块的内存不属于temp
}


// 全局变量和局部变量
// 一：局部变量： 在一个函数内部定义的变量，只能在函数内使用
	// 在main函数中的变量，只能在main函数中，即便main函数调用了其他函数
	// 不同函数当中的相同名字变量，互不相关
// 二：全局变量： 在函数外定义的变量，成为全局变量（外部变量）
	// 优点：不同的函数之间可以进行交互，而不需要形参实参的传递
	// 缺点：不便于阅读，全局变量会一直存在于系统当中，会占用内存

// extern int a; 表示在外部某个位置定义了变量a 这里不分配内存，只是做声明：后面有地方定义了a变量
int a;
// 1、外部变量的定义只有有一次，会分配内存，定义时可以进行初始化
// 2、外部变量的声明很有很多次，声明不会分配内存
// 3、在同一个源文件当中，如果出现的全局变量和局部变量同名，在局部变量的作用范围内，全局变量不起作用


// 变量的存储类别
// 从变量存在的时间（生存周期）角度来划分，可以将变量分为： 
		//  静态存储变量和动态存储变量，从而引出了静态存储方式和动态存储方式
		//  静态存储变量： 在程序运行期间分配固定存储空间的变量，这种分配变量的方式为静态存储方式
		//  动态存储变量： 在程序运行期间根据需要进行动态分配存储空间的变量，这种分配方式为动态存储方式

//全局变量（在函数的外部定义的）放在静态存储区中，程序开始执行的时候给全局变量分配存储区﹐程序执行完毕后释放这些存储区
//在程序执行过程中它们占据固定的存储单元，而不是动态的分配和释放。

//动态存储区中存哪些数据呢?
// a)函数形参，函数形参被看做是局部变量。
// b)局部变量，比如函数内定义的一些变量。
// c)函数调用时现场的一些数据和返回地址等。
// 一般来讲，这些数据在函数调用开始时分配存储空间﹐函数调用完毕，这些空间就祊释放掉了。这种分配和释放，我们就认为是动态的
// 如果你两次调用同—个函蠢，分配给此函数的局部变里等等的存储空间地址可能就是不同的。


// 局部变量的存储方式
// 1.传统情形，函数的局部变量，在函数被调用时分配内存，函数结束时释放内存
// 2.局部静态变量，用static修饰的变量
	// 其能够保留原值，占用的存储单元不释放，在下一次调用函数时，该值是上次调用结束时的值
void testStatic() {
	// 静态局部变量
	// 在静态存储区分配内存，程序整个运行期间不释放
	// 局部静态变量是在编译时赋予初值的，只赋予一次
	// 定义局部静态变量，不赋予初值会被赋值为0，普通的局部变量不赋予初值是一个不确定的值
	static int t = 10;	
	t++;
	printf("%d\n", t);
}


// 全局变量跨文件使用的方式
// 场景： 在a.cpp文件中定义的变量，在b.cpp文件当中使用
// 使用extern 做一个外部变量的说明（b.cpp文件的头部），说明这个变量是其他文件当中被定义过的外部变量，本文件不需要再进行申明
// 使用方法：在b.cpp头部写： extern int temp;
// 值得注意的是：在b.cpp文件中改变了变量的值，则在a.cpp文件中也会改变
// 如果只想在a.cpp文件当中使用变量temp.只需要在a.cpp文件： static int temp; 就无法在其他文件中使用

// 函数的跨文件使用
// 根据文件是否能被其他源文件使用，函数被分为内部函数和外部函数
// 在函数的申明前面加上static，表示为内部函数，只能在本文件当中使用
// 两个文件中的函数均加static，尽管在不同的文件当中出现同名函数，也不会影响

// 在函数前面加上extern 表示为外部函数 表示从外部来的函数

// 总结static的用法
// 1、在函数内部的一个变量前加上static，表示该变量只能存在于静态存储区
	  // 在函数编译的时候初始化，如果不给初值，则被赋予初值Wie0，并且下次调用时，维持上次调用结束时的值
// 2、在全局变量前加上static，表示该变量只能在本文件中使用，不能在其他文件中使用
// 3、在函数定义前面加static，表示该函数只能在本文件中使用，不能在其他文件中使用



int main() {
	// 测试递归函数
	printf("%d\n", getJieCheng(5));

	// 测试数组名作为函数的参数，地址传递的过程
	int tempA[2] = { 1,2 };
	testShuZuCanShu(tempA);
	printf("%d-%d\n", tempA[0], tempA[1]);	// 此时两个元素均变成了100

	// 测试程序块内的局部变量
	// 用大括号编写一段代码，大括号内的叫复合语句，{}内的变量只能在复合语句中使用，复合语句也叫程序块
	int a = 0; int b = 1;
	{
		int c = a + b;	// 程序块内可以使用main函数内定义的变量，a,b可以使用
	}
	// c = a + b; 程序显示未定义的标识符c 一旦离开程序块，c的内存就被释放了

	// 测试静态局部变量的存储
	testStatic(); // 输出三次为： 11 12 13
	testStatic();
	testStatic();


	return 0;
}
