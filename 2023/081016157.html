C++进阶07：类-inline-const-mutable-this-static

#include<iostream>
#include "temp10_类_time.h"
using namespace std;

static int global_a = 10;	// 保存在静态存储区，如果米有赋予初值，系统会给初值0，一般的变量则不会默认赋初值

void func() {
	static int a = 1;  // 局部静态变量： 程序只会在第一次执行时初始化为1，下一次执行函数时，会保持上次的结果

	a = 2;		// 当下一次程序调用时，a的值会保持为2
}

// 静态成员变量定义（分配内存）
int myTime::myStatic = 0;		// 可以不给初值，那么系统默认给0，定义时不需要给static，只能在一个头文件中定义


int main() {

	// 一、在类内实现成员函数inline
	// 直接在类中定义的成员函数，会被当做inline内联函数来处理
	// 内联函数： 系统将函数体内的代码直接执行，忽略调用的过程，提高程序的执行效率
	// 多个cpp文件可以直接include该头文件，因为系统会将其当做inline函数，不会看做是函数，而是代码块，不会出现函数重复定义的情况


	// 二、成员函数末尾的const 常量成员函数
	// const： 常量，在成员函数的后面加一个const，在成员函数的声明和定义中增加const
	// void Time::addHour(int number) const{}   表示在函数内部不会修改time对象的成员变量
	// 作用： 告诉系统，这个成员函数 不会修改该对象内任何成员变量的值等等
	// const只能放在成员函数的后面，表示不会修改成员变量的值


	// 三、mutable(不稳定，容易改变的意思)， const的反义词，mutable的引入是为了突破const的限制
	// 一个成员变量被mutable修饰了，那么它就一直处于可被修改的状态，即便被const修饰
	// mutable int hour; 那么在const末尾修饰的方法中就能被修改

	// 四、返回自身对象的引用： this
	// (1) this只能在成员函数中使用，全局函数，静态函数都不能使用this指针
	// (2) 在普通成员函数中，this是一个指向非const对象的const指针，this只能为指向该类的类型： myTime * const this.
		// this只能指向myTime对象

	myTime* m = new myTime(1,2,3);
	m->add_Hour(2);
	cout << m->Hour << endl;
	delete m;


	// 五、static成员
	// 属于整个类的成员变量，这种成员变量叫做static成员变量
	// 特点： 不属于某个对象，属于整个类，一旦在某个对象中修改了这个成员变量的值，在其他对象中能看到修改后的结果
	// 这种成员变量只有一个副本，对于这种成员变量的引用，					 使用  类名:: 成员变量名
	// 成员函数前面也可以加static构成： 静态成员函数， 属于整个类的成员函数，使用  类名:: 成员函数名(...)

	// 如何定义静态成员变量（分配内存-在类定义的头文件中没有给静态成员变量分配内存）
	// 答： 我们会在使用它的源文件开头位置定义，只能在一个源文件中定义，多个cpp文件定义会重复定义错误
	// 这种静态成员变量，在多个对象之间共同使用，跨对象使用
	cout << "类中的静态成员变量" << myTime::myStatic << endl;

	myTime* m2 = new myTime();
	myTime* m3 = new myTime();

	m2->myStatic = 20;
	cout << "类中的静态成员变量" << myTime::myStatic << endl;			// 一次修改，整个类的静态成员函数均为20
	cout << "myTime* m2 中的静态成员变量" << m2->myStatic << endl;	// 各个对象的该值都得到了修改
	cout << "myTime* m3 中的静态成员变量" << m3->myStatic << endl;	// 20
	delete m2, m3;
	// 上述的 myTime::myStatic  m2->myStatic  m3->myStatic  三个值指向的是同一块内存地址

	return 0;
}

myTime.h

#ifndef __MYTIME__
#define __MYTIME__

// 注意：当前头文件是myTime类的声明（定义）
// 情况：在1.cpp和2.cpp文件当中均#include<myTime.h>，生成时，为什么系统不会报错
//       但是当在这个文件中声明（定义）一个全局变量
// int global_temp1 = 0;	  // 声明 并且定义了变量     如果多个源文件引用了本头文件，会报错
// int global_temp(int);   // 仅仅声明，没有定义变量  如果多个源文件引用了本头文件，不会报错
// 答案： 这是因为遵守： “单一定义规则One-Definition Rule ODR”  根据此规则， 如果对同一个类的两个定义完全相同且出现在不同编译单位，会被当作同一个定义
// 类不会出现这种情况
// https://www.apiref.com/cpp-zh/cpp/language/definition.html
// https://blog.csdn.net/baoxiaofeicsdn/article/details/48338515

class myTime {
public:
	int Hour;
	int Minute;
	int Second;

	// 成员函数的定义，成员函数的内容在temp10_类_time.cpp文件中
	void initTime(int tempHour, int tempMinute, int tempSecond);

	// 用于访问私有成员变量的函数
	void initMillTime(int mls);

	// 构造函数  
	// 一般这种带一个参数的成员函数，声明为explicit，不允许隐式的类型转换
	explicit myTime();		// 无参构造函数
	myTime(int tempHour, int tempMinute, int tempSecond);		// 三个参数的构造函数

	// 拷贝构造函数
	// 一般而言，在拷贝构造函数的第一个参数前面加上const
	myTime(const myTime &tempTime, int a = 10);

	// 重载运算符=
	myTime&  operator=(const myTime& tempObj);

	// 析构函数（释放函数）
	~myTime();

private:
	int Millsecond;

public:

	// 在类的声明头文件当中，添加函数的实现
	// 这种在类的定义中实现的成员函数，会被当做inline内联函数处理
	void addHour(int number) {
		Hour += number;
	}

	// 成员函数中this的使用
	myTime& add_Hour(int number);

	//声明静态成员变量
	static int myStatic;		// 没有分配内存，不能在这里初始化
	
	// 声明静态成员函数
	static void myStaticFun(int number);
};

#endif

myTime.cpp

#include<iostream>
#include "temp10_类_time.h"
using namespace std;

// 成员函数initTime的实现部分
// 注意的是：这里需要添加类名myTime,限定作用域
//  ::  叫做  作用域运算符

void myTime::initTime(int tempHour, int tempMinute, int tempSecond) {
	// 在成员函数当中，可以直接使用成员变量名： Hour Minute Second
	Hour = tempHour;
	Minute = tempMinute;
	Second = tempSecond;
}

// 不管成员变量是否是私有的，成员函数都可以访问
void myTime::initMillTime(int mls) {
	Millsecond = mls;
}

// 构造函数的实现
myTime::myTime(int tempHour, int tempMinute, int tempSecond) {
	Hour = tempHour;
	Minute = tempMinute;
	Second = tempSecond;
	initMillTime(0);
}

// 无参构造函数，并且完成对成员变量的初始化
// 比较专业的写法： 在函数体之前写： 注意是用，分割开来
myTime::myTime() : Hour(12) , Minute(59) , Second(59) {
	initMillTime(99);
	std::cout << "调用了:myTime() : Hour(12) , Minute(59) , Second(59) 无参构造函数" <<std::endl;
}

// 静态成员函数的实现
// 静态成员函数不属于某个对象，是属于整个类的
void myTime::myStaticFun(int number) {
	// Hour = 10  // 错误，因为Hour是成员变量，是和对象相关的成员变量
	// 静态函数内部编写与类相关的,如静态成员变量
	//myStatic = number;
}

// 拷贝构造函数
// 第一个参数：必须是该类类型的引用，系统在特定的时机会自动调用
myTime::myTime(const myTime& tempTime, int a) {
	// 代码块： 拷贝过程
	std::cout << "调用了:myTime::myTime(const myTime& tempTime, int a) 拷贝构造函数" << std::endl;
}

// 返回的是对象本身，在调用完成了hour++之后，将this返回
myTime& myTime::add_Hour(int number) {
	Hour += number;
	return *this;
}

// 赋值运算符重载
myTime& myTime::operator=(const myTime& tempObj) {
	Hour = tempObj.Hour;
	Minute = tempObj.Minute;
	Second = tempObj.Second;
	std::cout << "调用了赋值运算符重载=" << std::endl;
	// 返回的是对象本身，所有使用return *this
	return *this;
}

// 析构函数（释放函数）
myTime::~myTime() {
	std::cout << "调用了析构函数" << std::endl;
}

一、在类内实现成员函数inline

直接在类中定义的成员函数，会被当做inline内联函数来处理
内联函数： 系统将函数体内的代码直接执行，忽略调用的过程，提高程序的执行效率
多个cpp文件可以直接include该头文件，因为系统会将其当做inline函数，不会看做是函数，而是代码块，不会出现函数重复定义的情况

二、成员函数末尾的const 常量成员函数

const： 常量，在成员函数的后面加一个const，在成员函数的声明和定义中增加const
void Time::addHour(int number) const{}   表示在函数内部不会修改time对象的成员变量
作用： 告诉系统，这个成员函数 不会修改该对象内任何成员变量的值等等
const只能放在成员函数的后面，表示不会修改成员变量的值

三、mutable(不稳定，容易改变的意思)， const的反义词，mutable的引入是为了突破const的限制

一个成员变量被mutable修饰了，那么它就一直处于可被修改的状态，即便被const修饰
mutable int hour; 那么在const末尾修饰的方法中就能被修改

四、返回自身对象的引用： this

(1) this只能在成员函数中使用，全局函数，静态函数都不能使用this指针
(2) 在普通成员函数中，this是一个指向非const对象的const指针，this只能为指向该类的类型： myTime * const this.
     this只能指向myTime对象

五、static成员

属于整个类的成员变量，这种成员变量叫做static成员变量

特点： 不属于某个对象，属于整个类，一旦在某个对象中修改了这个成员变量的值，在其他对象中能看到修改后的结果

这种成员变量只有一个副本，对于这种成员变量的引用， 使用 类名:: 成员变量名
成员函数前面也可以加static构成： 静态成员函数， 属于整个类的成员函数，使用 类名:: 成员函数名(…)

如何定义静态成员变量（分配内存-在类定义的头文件中没有给静态成员变量分配内存）
答： 我们会在使用它的源文件开头位置定义，只能在一个源文件中定义，多个cpp文件定义会重复定义错误

// 这种静态成员变量，在多个对象之间共同使用，跨对象使用
cout << "类中的静态成员变量" << myTime::myStatic << endl;

myTime* m2 = new myTime();
myTime* m3 = new myTime();

m2->myStatic = 20;
cout << "类中的静态成员变量" << myTime::myStatic << endl;			// 一次修改，整个类的静态成员函数均为20
cout << "myTime* m2 中的静态成员变量" << m2->myStatic << endl;	// 各个对象的该值都得到了修改
cout << "myTime* m3 中的静态成员变量" << m3->myStatic << endl;	// 20
delete m2, m3;
// 上述的 myTime::myStatic  m2->myStatic  m3->myStatic  三个值指向的是同一块内存地址

六、默认的构造函数

(1) 没有参数的构造函数，被称为 默认构造函数
(2) 在创建对象时，会调用构造函数，当我们没有在类中定义构造函数时，系统会调用默认构造函数（没有参数的构造函数） 隐式的自动定义一个默认构造函数
(3) 一旦我们自己定义了构造函数，系统就不会创建这个 默认的构造函数了

七、=default =delete

// =default 编译器能够为我们自动生成函数体, 一般的函数，带有参数的函数，都不能使用
// myTime(){} 等价于 myTime() = default;
// =delete
// myTime = delete; 让程序显式的禁用某个函数

八、拷贝构造函数

默认情况下，类对象的拷贝是每个成员变量依次拷贝
myTime myTime1; // 调用默认构造函数,并且初始化了对象

拷贝构造函数
如果一个类的构造函数的第一个参数 是所属的类类型的引用，如果还有其他额外的参数
这些额外的参数还有默认值，则这个构造函数也叫： 拷贝构造函数
函数的默认参数必须放在函数的声明中，除非该函数没有函数声明

注意：
(1) 建议拷贝构造函数第一个参数总是带着const： 第一个参数是所属类的类型的引用，加上const表示拷贝过程中不会不小心更改了
(2) 拷贝构造函数不建议声明成： explicit ： 如果声明了，会发现上述4种情况1,4会报错，因为这是隐式的写法

// 调用拷贝构造函数的时机
myTime myTime2 = myTime1;	// 调用了拷贝构造函数，4个相同
myTime myTime3(myTime1);
myTime myTime4{ myTime1 };
myTime myTime5 = { myTime1 };

注意：
(1) 如果我们没有定义一个拷贝构造函数，编译器会帮我们定义一个
(2) 如果是编译器帮我们定义的拷贝构造函数，这个拷贝构造函数也是将对象的成员依次拷贝到正在创建的对象中去
